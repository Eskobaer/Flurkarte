  <script>
    function zeigeHinweis() {
      const adresse = document.getElementById("adresse").value.trim();
      if (!adresse) { alert("Bitte eine Adresse eingeben."); return; }
      const street = adresse.split(',')[0].replace(/\\s+/g,'_');
      const pdfName = `Flurkarte_${street}.pdf`;

      alert(
        "Du wirst zur offiziellen Bestellung weitergeleitet.\\n\\n" +
        "Bitte dort dein Flurstück suchen und bestellen.\\n\\n" +
        "PDF später idealerweise speichern als:\\n" + pdfName
      );

      window.open("https://gds.hessen.de/produkte", "_blank");
    }
  </script>
